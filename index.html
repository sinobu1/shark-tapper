  <script>
    document.addEventListener("DOMContentLoaded", function() {
      const dbUrl = "https://tralalero-fec07-default-rtdb.firebaseio.com";
      let userId = null;
      let userData = {
        username: "Player",
        avatar: "https://placehold.co/64x64",
        tokens: 0,
        skin: 'default'
      };
      let lastSave = 0;
      const SAVE_INTERVAL = 5000;
      let canVibrate = 'vibrate' in navigator;
      const loader = document.getElementById('loader');
      let loaderTimeout;

      // Initialize app
      function init() {
        setupEventListeners();
        loadTelegram();
        initLocalStorage();
        showTab('home');
      }

      // Event listeners setup
      function setupEventListeners() {
        // Shark tap
        const tapImage = document.getElementById('tapImage');
        tapImage.addEventListener('click', handleTap);
        
        // Skin selection
        document.querySelectorAll('.shark-gallery img').forEach(img => {
          img.addEventListener('click', function() {
            setSkin(this.dataset.skin);
          });
        });
        
        // Navigation
        document.querySelectorAll('nav button').forEach(btn => {
          btn.addEventListener('click', function() {
            showTab(this.dataset.tab);
          });
        });
      }

      // Handle tap on shark
      function handleTap(event) {
        // Vibrate on tap if supported
        if (canVibrate) {
          navigator.vibrate(10);
        }
        
        // Animation
        const shark = document.getElementById("tapImage");
        shark.style.transform = "scale(0.95)";
        setTimeout(() => shark.style.transform = "scale(1)", 100);
        
        createTapEffect(event);
        createCoinPopup(event);
        
        // Increment coins
        userData.tokens++;
        updateDisplay();
        
        // Save progress
        const now = Date.now();
        if (now - lastSave > SAVE_INTERVAL) {
          saveProgress();
          lastSave = now;
        } else {
          saveToLocalStorage();
        }
      }

      // Create tap visual effect
      function createTapEffect(event) {
        const effect = document.createElement('div');
        effect.className = 'tap-effect';
        effect.style.left = `${event.clientX}px`;
        effect.style.top = `${event.clientY}px`;
        document.body.appendChild(effect);
        
        setTimeout(() => {
          effect.remove();
        }, 600);
      }

      // Create +1 coin popup
      function createCoinPopup(event) {
        const popup = document.createElement('div');
        popup.className = 'coin-popup';
        popup.textContent = '+1';
        popup.style.left = `${event.clientX}px`;
        popup.style.top = `${event.clientY - 20}px`;
        document.body.appendChild(popup);
        
        setTimeout(() => {
          popup.remove();
        }, 1000);
      }

      // Update UI display
      function updateDisplay() {
        const counter = document.getElementById("profileTokens");
        if (counter) counter.textContent = userData.tokens.toLocaleString();
        
        const levelEl = document.getElementById("profileLevel");
        if (levelEl) levelEl.textContent = `Coins: ${userData.tokens.toLocaleString()}`;
        
        const bar = document.getElementById("xpBar");
        if (bar) {
          const limits = [0, 100, 500, 1000, 2500, 5000, 10000];
          let max = limits.find(l => userData.tokens < l) || 10000;
          let min = limits[limits.indexOf(max) - 1] || 0;
          let percent = ((userData.tokens - min) / (max - min)) * 100;
          bar.style.width = `${Math.min(100, Math.max(0, percent))}%`;
        }
      }

      // Show tab content
      function showTab(tabId) {
        // Update nav buttons
        document.querySelectorAll('nav button').forEach(btn => {
          btn.classList.toggle('active', btn.dataset.tab === tabId);
        });
        
        // Show selected tab
        document.querySelectorAll('.tab-page').forEach(page => {
          page.classList.toggle('active', page.id === tabId);
        });
        
        // Load leaderboard if needed
        if (tabId === 'leaderboard') {
          loadLeaderboard();
        }
      }

      // Save progress to server and local storage
      async function saveProgress() {
        showLoader();
        try {
          if (!userId) {
            saveToLocalStorage();
            return;
          }
          
          const response = await fetch(`${dbUrl}/users/${userId}.json`, {
            method: "PUT",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({
              balance: userData.tokens,
              first_name: userData.username,
              photo_url: userData.avatar,
              skin: userData.skin,
              last_update: Date.now()
            })
          });
          
          if (!response.ok) throw new Error('Save failed');
          saveToLocalStorage();
        } catch(err) {
          console.error('Save error:', err);
          saveToLocalStorage();
        } finally {
          hideLoader();
        }
      }

      // Load from local storage
      function initLocalStorage() {
        const savedData = localStorage.getItem('sharkTapperData');
        if (savedData) {
          try {
            const parsed = JSON.parse(savedData);
            userData.tokens = parsed.tokens || 0;
            userData.skin = parsed.skin || 'default';
            updateDisplay();
            setSkin(userData.skin, true);
          } catch(e) {
            console.error('Error parsing saved data', e);
          }
        }
      }

      // Save to local storage
      function saveToLocalStorage() {
        localStorage.setItem('sharkTapperData', JSON.stringify({
          tokens: userData.tokens,
          skin: userData.skin
        }));
      }

      // Load progress from server
      async function loadProgress() {
        showLoader();
        try {
          if (!userId) return;
          
          const response = await fetch(`${dbUrl}/users/${userId}.json`);
          if (!response.ok) throw new Error('Network response was not ok');
          
          const data = await response.json();
          if (data) {
            // Use highest value between server and local
            const serverTokens = data.balance || 0;
            userData.tokens = Math.max(serverTokens, userData.tokens);
            
            // Update user data
            userData.username = data.first_name || "Player";
            userData.avatar = data.photo_url || "https://placehold.co/64x64";
            userData.skin = data.skin || 'default';
            
            // Update UI
            updateDisplay();
            document.getElementById("profileName").textContent = userData.username;
            document.getElementById("profileAvatar").src = userData.avatar;
            setSkin(userData.skin, true);
            
            // Save merged data
            saveToLocalStorage();
          }
        } catch(err) {
          console.error('Load error:', err);
        } finally {
          hideLoader();
        }
      }

      // Load leaderboard
      async function loadLeaderboard() {
        showLoader();
        try {
          const response = await fetch(`${dbUrl}/users.json?orderBy="balance"&limitToLast=10`);
          if (!response.ok) throw new Error('Network response was not ok');
          
          const data = await response.json();
          const list = document.getElementById("leaderboardList");
          
          if (!data || Object.keys(data).length === 0) {
            list.innerHTML = '<li style="text-align:center; color: var(--subtext);">No players yet</li>';
            return;
          }
          
          // Sort by balance
          const sorted = Object.entries(data)
            .sort((a, b) => (b[1].balance || 0) - (a[1].balance || 0))
            .slice(0, 10);
          
          renderLeaderboard(sorted);
        } catch(err) {
          console.error('Leaderboard error:', err);
          const list = document.getElementById("leaderboardList");
          list.innerHTML = '<li style="text-align:center; color: var(--subtext);">Loading error</li>';
        } finally {
          hideLoader();
        }
      }

      // Render leaderboard
      function renderLeaderboard(data) {
        const list = document.getElementById("leaderboardList");
